import { Component, Inject, OnInit } from '@angular/core';
import { FormControl, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';

@Component({
  selector: 'app-query1',
  templateUrl: './query1.component.html',
  styleUrls: ['./query1.component.css']
})
export class Query1Component implements OnInit {

  constructor(public dialogRef: MatDialogRef<Query1Component>,
              @Inject(MAT_DIALOG_DATA) public data: any) { }

  ngOnInit(): void {
  }

  submit(): void {
  }

  onNoClick(): void {
    this.dialogRef.close();
  }

  save(): void {
  }

  integerControlMin = new FormControl('', [Validators.required, Validators.pattern(/^\d{1,6}$/)]);
  integerControlMax = new FormControl('', [Validators.required, Validators.pattern(/^\d{1,6}$/)]);

  getErrorMessageIntegerMin() {
    return this.integerControlMin.errors['pattern'] ? 'Only numbers' : 
            this.integerControlMin.errors['required'] ? 'Required Minimum Value': '';
  }

  getErrorMessageIntegerMax() {
    return this.integerControlMax.errors['pattern'] ? 'Only numbers' : 
            this.integerControlMax.errors['required'] ? 'Required Maximum Value': '';
  }

}


<div class="container">
    <form class="mat-dialog-content" (ngSubmit)="submit" #formControl="ngForm">

    <div class="form">
        <mat-form-field color="accent">
            <input matInput #input [formControl]="integerControlMin" class="form-control" maxlength="6" [(ngModel)]="data.minValue" placeholder="Minimum value" name="minValue" required>
            <mat-error *ngIf="integerControlMin.invalid">{{getErrorMessageIntegerMin()}}</mat-error>
        </mat-form-field>
    </div>    

    <div class="form">
        <mat-form-field color="accent">
            <input matInput #input [formControl]="integerControlMax" class="form-control" maxlength="6" [(ngModel)]="data.maxValue" placeholder="Maximum value" name="maxValue" required>
            <mat-error *ngIf="integerControlMax.invalid">{{getErrorMessageIntegerMax()}}</mat-error>
        </mat-form-field>
    </div>    

    <div mat-dialog-actions>
        <button mat-button type="submit" [disabled]="!integerControlMin.valid || !integerControlMax.valid" [mat-dialog-close]="1" (click)="save()">Save</button>
        <button mat-button (click)="onNoClick()" tabindex="-1">Cancel</button>  
      </div>
    </form>
  </div>
